Пример работы плагина (rus)

$('.my_tiles').showMeMore({
    current: 3,
    count: 3,
    showButtonText: 'show me more',
    hideButtonText: 'hide all tiles',
    enableHide: true
});

current: 3 - показать при старте три елемента

count: 3 - по сколько элементов показывать при клике

showButtonText: 'show me more' - текст кнопки для показа следующих елементов

enableHide: true - включаем функцию "спрятать елементы" (кнопка показывается после того как будут отображены все элементы)

$('.my_tiles_1').showMeMore({
    current: 6,
    count: 3,
    fadeSpeed: 1000,
    showButton: '.showBtn',
    hideButton: '.hideBtn',
    list: '.tile_item'
});

fadeSpeed: 1000 - скорость показа итема (ms)

showButton: '.showBtn' - передаем плагину свою кнопку "показать" (отключает генерацию стандартной)

hideButton: '.hideBtn' - передаем плагину свою кнопку "спрятать" (отключает генерацию стандартной)

list: '.tile_item' - указываем элементы для плагина внутри контейнера
